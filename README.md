# bilian-calling-for-tenders_scrapy
使用scrapy爬取必联网招标信息，使用代理IP池，并存入mysql数据库中;
信息存储字段：
projectcode = '',  # 项目编号
web = '',  # 信息来源网站
keyword = '',  # 关键字
detail_url = '',  # 招标详细页网址
title = '',  # 第三方网站发布标题
toptype = '',  # 信息类型
province = '',  # 归属省份
product = '',  # 产品范畴
industry = '',  # 归属行业
tendering_manner = '',  # 招标方式
publicity_date = '',  # 招标公示日期
expiry_date = '',  # 招标截止时间
